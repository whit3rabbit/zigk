**In `src/uapi/process/signal.zig`**, add a user-visible `SigInfoT` struct (the struct pushed to user stack):

```zig
/// User-visible siginfo_t structure pushed onto user stack for SA_SIGINFO handlers.
/// Must be 128 bytes to match Linux ABI.
///
/// Layout (offsets in bytes):
///   0-3:   si_signo (i32)
///   4-7:   si_errno (i32)
///   8-11:  si_code  (i32)
///  12-15:  _pad0    (i32) -- padding / __si_fields union discriminator
///  16-19:  si_pid   (i32)
///  20-23:  si_uid   (i32)
///  24-27:  si_status (i32)
///  28-31:  _pad1    (i32)
///  32-35:  si_value_int (i32)
///  36-39:  _pad2    (i32) -- alignment padding before pointer
///  40-47:  si_value_ptr (usize, 8 bytes on 64-bit)
///  48-127: _pad     (80 bytes remaining padding to reach 128 total)
pub const SigInfoT = extern struct {
    si_signo: i32,      // Signal number
    si_errno: i32,      // Error number (0)
    si_code: i32,       // Signal code (SI_USER, SI_QUEUE, SI_KERNEL, SI_TKILL)
    _pad0: i32 = 0,     // Padding (or __si_fields union discriminator)
    si_pid: i32,        // Sender PID
    si_uid: i32,        // Sender UID
    si_status: i32 = 0, // Exit status (for SIGCHLD)
    _pad1: i32 = 0,
    si_value_int: i32 = 0,  // si_value.sival_int
    _pad2: i32 = 0,         // Alignment padding
    si_value_ptr: usize = 0, // si_value.sival_ptr (8 bytes on 64-bit)
    _pad: [128 - 48]u8 = [_]u8{0} ** (128 - 48),

    comptime {
        const std = @import("std");
        std.debug.assert(@sizeOf(SigInfoT) == 128);
    }
};
```

**CRITICAL SIZE NOTE:** The named fields before `_pad` consume exactly 48 bytes (not 40). The last named field `si_value_ptr` is `usize` (8 bytes) ending at offset 48. Therefore `_pad` must be `[128 - 48]u8 = [80]u8`, NOT `[128 - 40]u8 = [88]u8`. Using 88 bytes of padding would make the struct 136 bytes and the comptime assert would fail to compile.

**Test 18: SA_SIGINFO handler receives correct siginfo_t with si_pid**

This test installs a SA_SIGINFO handler for SIGUSR1, sends a signal to self via kill(), and verifies the handler receives correct si_signo and si_pid in the siginfo_t.

```zig
// Shared state for SA_SIGINFO handler tests
var siginfo_received_signo: i32 = 0;
var siginfo_received_code: i32 = 0;
var siginfo_received_pid: i32 = 0;
var siginfo_handler_called: bool = false;

// SA_SIGINFO handler: receives (signum, *siginfo_t, *ucontext_t)
fn handleSigusr1WithInfo(sig: i32, info_ptr: usize, _ucontext: usize) callconv(.c) void {
    _ = _ucontext;
    _ = sig;
    siginfo_handler_called = true;
    if (info_ptr != 0) {
        // Read siginfo_t fields from the pointer
        // siginfo_t layout: si_signo (i32@0), si_errno (i32@4), si_code (i32@8),
        //                   _pad (i32@12), si_pid (i32@16), si_uid (i32@20)
        const info: *const syscall.SignalSigInfo = @ptrFromInt(info_ptr);
        siginfo_received_signo = info.si_signo;
        siginfo_received_code = info.si_code;
        siginfo_received_pid = info.si_pid;
    }
}

// Test 18: SA_SIGINFO handler receives correct metadata from kill()
pub fn testSiginfoPidUid() !void {
    const SIGUSR1 = 10;
    const SA_SIGINFO: u64 = 0x00000004;

    // Reset state
    siginfo_handler_called = false;
    siginfo_received_signo = 0;
    siginfo_received_code = 0;
    siginfo_received_pid = 0;

    // Install SA_SIGINFO handler
    var act = syscall.SigAction{
        .handler = @intFromPtr(&handleSigusr1WithInfo),
        .flags = SA_SIGINFO,
        .restorer = 0,
        .mask = 0,
    };
    try syscall.sigaction(SIGUSR1, &act, null);

    // Send signal to self via kill()
    const my_pid = syscall.getpid();
    try syscall.kill(my_pid, SIGUSR1);

    // Give handler time to execute
    syscall.sched_yield() catch {};
    syscall.sched_yield() catch {};

    // Verify handler was called
    if (!siginfo_handler_called) return error.TestFailed;

    // Verify siginfo fields
    if (siginfo_received_signo != SIGUSR1) return error.TestFailed;
    // si_code should be SI_USER (0) since sent via kill()
    if (siginfo_received_code != 0) return error.TestFailed;
    // si_pid should be our own PID
    if (siginfo_received_pid != @as(i32, @intCast(my_pid))) return error.TestFailed;

    // Restore default handler
    var default_act = syscall.SigAction{
        .handler = @intFromPtr(&handleSigusr1),
        .flags = 0,
        .restorer = 0,
        .mask = 0,
    };
    _ = syscall.sigaction(SIGUSR1, &default_act, null) catch {};
}
```

**Test 19: rt_sigqueueinfo metadata survives to rt_sigtimedwait**

This test validates that siginfo data sent via rt_sigqueueinfo can be read back via rt_sigtimedwait with correct si_code and si_pid.

```zig
// Test 19: rt_sigqueueinfo metadata preserved in rt_sigtimedwait
pub fn testSiginfoQueueRoundTrip() !void {
    const SIGUSR2 = 12;
    const SIG_BLOCK = 0;
    const SIG_UNBLOCK = 1;

    // Block SIGUSR2 so it stays pending
    var set: syscall.SigSet = 0;
    syscall.uapi.signal.sigaddset(&set, SIGUSR2);
    try syscall.sigprocmask(SIG_BLOCK, &set, null);

    // Send signal with SI_QUEUE code and known PID
    const my_pid = syscall.getpid();
    var send_info = syscall.SignalSigInfo{
        .si_signo = SIGUSR2,
        .si_code = syscall.SI_QUEUE, // -1
        .si_pid = @intCast(my_pid),
        .si_uid = 0,
    };

    syscall.rt_sigqueueinfo(@intCast(my_pid), SIGUSR2, &send_info) catch |err| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        if (err == error.NotImplemented) return error.SkipTest;
        return err;
    };

    // Dequeue via rt_sigtimedwait
    var timeout = syscall.SignalTimespec{ .tv_sec = 0, .tv_nsec = 0 };
    var recv_info: syscall.SignalSigInfo = .{};
    const signo = syscall.rt_sigtimedwait(&set, &recv_info, &timeout) catch |err| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        if (err == error.NotImplemented) return error.SkipTest;
        return err;
    };

    // Unblock
    _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};

    // Verify signal number
    if (signo != SIGUSR2) return error.TestFailed;

    // Verify siginfo fields from queue
    if (recv_info.si_signo != @as(i32, SIGUSR2)) return error.TestFailed;
    if (recv_info.si_code != syscall.SI_QUEUE) return error.TestFailed;
    if (recv_info.si_pid != @as(i32, @intCast(my_pid))) return error.TestFailed;
}
```

**Test 20: Standard signal coalescing (second send while pending is a no-op)**

```zig
// Test 20: Standard signal coalescing (second send while pending is a no-op)
pub fn testSiginfoStandardCoalescing() !void {
    const SIGUSR1 = 10;
    const SIG_BLOCK = 0;
    const SIG_UNBLOCK = 1;

    // Block SIGUSR1
    var set: syscall.SigSet = 0;
    syscall.uapi.signal.sigaddset(&set, SIGUSR1);
    try syscall.sigprocmask(SIG_BLOCK, &set, null);

    // Send SIGUSR1 twice while blocked
    try syscall.kill(syscall.getpid(), SIGUSR1);
    try syscall.kill(syscall.getpid(), SIGUSR1);

    // Dequeue via rt_sigtimedwait -- should get exactly one
    var timeout = syscall.SignalTimespec{ .tv_sec = 0, .tv_nsec = 0 };
    _ = syscall.rt_sigtimedwait(&set, null, &timeout) catch |err| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        if (err == error.NotImplemented) return error.SkipTest;
        return err;
    };

    // Second dequeue should fail (no more pending)
    const result2 = syscall.rt_sigtimedwait(&set, null, &timeout);
    if (result2) |_| {
        // Should NOT succeed -- signal was consumed
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        return error.TestFailed;
    } else |err| {
        if (err != error.WouldBlock) {
            _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
            if (err == error.NotImplemented) return error.SkipTest;
            return error.TestFailed;
        }
    }

    _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
}
```

**Test 21: RT signal queuing (multiple instances of same RT signal are queued and delivered in order)**

This test validates that real-time signals (32+) do NOT coalesce -- sending SIGRTMIN twice while blocked produces two deliveries via rt_sigtimedwait. This is the key behavioral difference between standard and RT signals.

```zig
// Test 21: RT signal queuing -- multiple instances of same RT signal are queued
pub fn testSiginfoRtSignalQueuing() !void {
    const SIGRTMIN = 32;
    const SIG_BLOCK = 0;
    const SIG_UNBLOCK = 1;

    // Block SIGRTMIN so signals stay pending
    var set: syscall.SigSet = 0;
    syscall.uapi.signal.sigaddset(&set, SIGRTMIN);
    try syscall.sigprocmask(SIG_BLOCK, &set, null);

    // Send SIGRTMIN twice via rt_sigqueueinfo with distinct si_value
    const my_pid = syscall.getpid();

    var send_info1 = syscall.SignalSigInfo{
        .si_signo = SIGRTMIN,
        .si_code = syscall.SI_QUEUE, // -1
        .si_pid = @intCast(my_pid),
        .si_uid = 0,
    };
    syscall.rt_sigqueueinfo(@intCast(my_pid), SIGRTMIN, &send_info1) catch |err| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        if (err == error.NotImplemented) return error.SkipTest;
        return err;
    };

    var send_info2 = syscall.SignalSigInfo{
        .si_signo = SIGRTMIN,
        .si_code = syscall.SI_QUEUE,
        .si_pid = @intCast(my_pid),
        .si_uid = 0,
    };
    syscall.rt_sigqueueinfo(@intCast(my_pid), SIGRTMIN, &send_info2) catch |err| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        if (err == error.NotImplemented) return error.SkipTest;
        return err;
    };

    // Dequeue first instance via rt_sigtimedwait -- should succeed
    var timeout = syscall.SignalTimespec{ .tv_sec = 0, .tv_nsec = 0 };
    var recv_info1: syscall.SignalSigInfo = .{};
    const signo1 = syscall.rt_sigtimedwait(&set, &recv_info1, &timeout) catch |err| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        if (err == error.NotImplemented) return error.SkipTest;
        return err;
    };
    if (signo1 != SIGRTMIN) {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        return error.TestFailed;
    }

    // Dequeue second instance -- should ALSO succeed (RT signals queue, not coalesce)
    var recv_info2: syscall.SignalSigInfo = .{};
    const signo2 = syscall.rt_sigtimedwait(&set, &recv_info2, &timeout) catch |err| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        // If EAGAIN/WouldBlock, RT signal queuing is broken -- only one was delivered
        if (err == error.WouldBlock) return error.TestFailed;
        if (err == error.NotImplemented) return error.SkipTest;
        return err;
    };
    if (signo2 != SIGRTMIN) {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        return error.TestFailed;
    }

    // Third dequeue should fail (only sent two)
    const result3 = syscall.rt_sigtimedwait(&set, null, &timeout);
    if (result3) |_| {
        _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
        return error.TestFailed;
    } else |_| {
        // Expected: WouldBlock or similar -- no more pending
    }

    _ = syscall.sigprocmask(SIG_UNBLOCK, &set, null) catch {};
}
```
