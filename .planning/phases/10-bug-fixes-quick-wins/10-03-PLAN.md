pub fn sys_getrlimit(resource: usize, rlim_ptr: usize) SyscallError!usize {
    if (resource >= uapi.RLIM_NLIMITS) return error.EINVAL;

    const proc = base.getCurrentProcess();
    const limit = proc.rlimits[resource];

    // Validate userspace buffer
    if (!isValidUserAccess(rlim_ptr, @sizeOf(uapi.Rlimit), .Write)) {
        return error.EFAULT;
    }

    UserPtr.from(rlim_ptr).writeValue(limit) catch return error.EFAULT;
    return 0;
}

```zig
pub fn sys_setrlimit(resource: usize, rlim_ptr: usize) SyscallError!usize {
    if (resource >= uapi.RLIM_NLIMITS) return error.EINVAL;

    // Read new limit from userspace
    if (!isValidUserAccess(rlim_ptr, @sizeOf(uapi.Rlimit), .Read)) {
        return error.EFAULT;
    }
    const new_limit = UserPtr.from(rlim_ptr).readValue(uapi.Rlimit) catch return error.EFAULT;

    // Validate: soft <= hard
    if (new_limit.soft > new_limit.hard) return error.EINVAL;

    const proc = base.getCurrentProcess();
    const old_limit = proc.rlimits[resource];

    // Permission check: raising hard limit requires CAP_SYS_RESOURCE (root)
    if (new_limit.hard > old_limit.hard and proc.euid != 0) {
        return error.EPERM;
    }

    // Store new limit
    proc.rlimits[resource] = new_limit;
    return 0;
}
```

```zig
pub fn sys_sigaltstack(ss_ptr: usize, old_ss_ptr: usize) SyscallError!usize {
    const thread = sched.getCurrentThread() orelse return error.ESRCH;

    // Return current stack if requested
    if (old_ss_ptr != 0) {
        if (!isValidUserAccess(old_ss_ptr, @sizeOf(uapi.StackT), .Write)) {
            return error.EFAULT;
        }
        const old_ss = uapi.StackT{
            .sp = thread.sigaltstack_sp,
            .flags = thread.sigaltstack_flags,
            .size = thread.sigaltstack_size,
        };
        UserPtr.from(old_ss_ptr).writeValue(old_ss) catch return error.EFAULT;
    }

    // Set new stack if provided
    if (ss_ptr != 0) {
        if (!isValidUserAccess(ss_ptr, @sizeOf(uapi.StackT), .Read)) {
            return error.EFAULT;
        }
        const new_ss = UserPtr.from(ss_ptr).readValue(uapi.StackT) catch return error.EFAULT;

        // Validate: cannot change stack while on it (SS_ONSTACK)
        if ((thread.sigaltstack_flags & uapi.SS_ONSTACK) != 0) {
            return error.EPERM;
        }

        // Store new stack
        thread.sigaltstack_sp = new_ss.sp;
        thread.sigaltstack_size = new_ss.size;
        thread.sigaltstack_flags = new_ss.flags & uapi.SS_DISABLE; // Only SS_DISABLE is settable
    }

    return 0;
}
```

- SS_DISABLE = 2 (alternate stack is disabled)

fn testSigaltstack() !void {
    const syscall = @import("syscall");

    // Allocate stack (8KB)
    const stack_size = 8192;
    const stack = try syscall.mmap(0, stack_size, syscall.PROT_READ | syscall.PROT_WRITE,
                                    syscall.MAP_PRIVATE | syscall.MAP_ANONYMOUS, -1, 0);
    defer _ = syscall.munmap(stack, stack_size) catch {};

    var new_ss: syscall.StackT = .{
        .sp = stack,
        .flags = 0,
        .size = stack_size,
    };
    var old_ss: syscall.StackT = undefined;

    // Set alternate stack
    try syscall.sigaltstack(&new_ss, &old_ss);

    // Read back
    var current_ss: syscall.StackT = undefined;
    try syscall.sigaltstack(null, &current_ss);

    // Verify it matches
    if (current_ss.sp != stack or current_ss.size != stack_size) {
        return error.TestFailed;
    }
}
