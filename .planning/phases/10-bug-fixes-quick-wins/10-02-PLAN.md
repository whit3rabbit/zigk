```zig
pub fn sys_dup3(oldfd: usize, newfd: usize, flags: usize) SyscallError!usize {
    // POSIX: dup3 returns EINVAL if oldfd == newfd
    if (oldfd == newfd) return error.EINVAL;

    // Validate flags
    if ((flags & ~@as(usize, fd_mod.O_CLOEXEC)) != 0) return error.EINVAL;

    const result = try sys_dup2(oldfd, newfd);

    // Apply O_CLOEXEC if requested
    if ((flags & fd_mod.O_CLOEXEC) != 0) {
        const table = base.getGlobalFdTable();
        const fd_u32 = safeFdCast(result) orelse return error.EBADF;
        if (table.get(fd_u32)) |fd| {
            fd.cloexec = true;
        }
    }

    return result;
}
```

```zig
    const valid_flags = uapi.socket.SOCK_NONBLOCK | uapi.socket.SOCK_CLOEXEC;
    if ((flags & ~valid_flags) != 0) return error.EINVAL;

    // Perform accept (returns new socket FD number)
    const new_fd = try acceptInternal(fd, addr_ptr, addrlen_ptr);
```
