   pub const SemidDs = uapi.ipc.sysv.SemidDs;
   pub const SemBuf = uapi.ipc.sysv.SemBuf;

   // ========== Shared Memory ==========

   pub fn shmget(key: i32, size: usize, shmflg: i32) SyscallError!u32 {
       const ret = syscall3(
           uapi.syscalls.SYS_SHMGET,
           @as(usize, @bitCast(@as(isize, key))),
           size,
           @as(usize, @bitCast(@as(isize, shmflg))),
       );
       if (isError(ret)) return errorFromReturn(ret);
       return @intCast(ret);
   }

   pub fn shmat(shmid: u32, shmaddr: ?[*]u8, shmflg: u32) SyscallError![*]u8 {
       const addr: usize = if (shmaddr) |a| @intFromPtr(a) else 0;
       const ret = syscall3(
           uapi.syscalls.SYS_SHMAT,
           @as(usize, shmid),
           addr,
           @as(usize, shmflg),
       );
       if (isError(ret)) return errorFromReturn(ret);
       return @ptrFromInt(ret);
   }

   pub fn shmdt(shmaddr: [*]const u8) SyscallError!void {
       const ret = syscall1(
           uapi.syscalls.SYS_SHMDT,
           @intFromPtr(shmaddr),
       );
       if (isError(ret)) return errorFromReturn(ret);
   }

   // Note: syscall1 may not exist; check primitive.zig. If not, use syscall2 with dummy 0 arg,
   // or use syscall3 with extra zeros. The kernel only reads arg1.
   // Actually, shmdt takes 1 arg. Check if there's a syscall1 function.
   // If not, wrap with: const ret = @as(isize, @bitCast(primitive.raw_syscall(SYS_SHMDT, addr, 0, 0, 0, 0, 0)));
   // The simplest approach: use syscall3 and pass 0 for unused args:
   // Actually the dispatch table passes all 6 args regardless, so any syscallN works.
   // Use the lowest available -- if syscall1 doesn't exist, use syscall2.

   pub fn shmctl(shmid: u32, cmd: i32, buf: ?*ShmidDs) SyscallError!usize {
       const buf_addr: usize = if (buf) |b| @intFromPtr(b) else 0;
       const ret = syscall3(
           uapi.syscalls.SYS_SHMCTL,
           @as(usize, shmid),
           @as(usize, @bitCast(@as(isize, cmd))),
           buf_addr,
       );
       if (isError(ret)) return errorFromReturn(ret);
       return ret;
   }

   // ========== Semaphores ==========

   pub fn semget(key: i32, nsems: u32, semflg: i32) SyscallError!u32 {
       const ret = syscall3(
           uapi.syscalls.SYS_SEMGET,
           @as(usize, @bitCast(@as(isize, key))),
           @as(usize, nsems),
           @as(usize, @bitCast(@as(isize, semflg))),
       );
       if (isError(ret)) return errorFromReturn(ret);
       return @intCast(ret);
   }

   pub fn semop(semid: u32, sops: []const SemBuf) SyscallError!void {
       const ret = syscall3(
           uapi.syscalls.SYS_SEMOP,
           @as(usize, semid),
           @intFromPtr(sops.ptr),
           sops.len,
       );
       if (isError(ret)) return errorFromReturn(ret);
   }

   pub fn semctl(semid: u32, semnum: u32, cmd: i32, arg: usize) SyscallError!usize {
       const ret = syscall4(
           uapi.syscalls.SYS_SEMCTL,
           @as(usize, semid),
           @as(usize, semnum),
           @as(usize, @bitCast(@as(isize, cmd))),
           arg,
       );
       if (isError(ret)) return errorFromReturn(ret);
       return ret;
   }

   // ========== Message Queues ==========

   pub fn msgget(key: i32, msgflg: i32) SyscallError!u32 {
       const ret = syscall2(
           uapi.syscalls.SYS_MSGGET,
           @as(usize, @bitCast(@as(isize, key))),
           @as(usize, @bitCast(@as(isize, msgflg))),
       );
       if (isError(ret)) return errorFromReturn(ret);
       return @intCast(ret);
   }

   pub fn msgsnd(msqid: u32, msgp: [*]const u8, msgsz: usize, msgflg: i32) SyscallError!void {
       const ret = syscall4(
           uapi.syscalls.SYS_MSGSND,
           @as(usize, msqid),
           @intFromPtr(msgp),
           msgsz,
           @as(usize, @bitCast(@as(isize, msgflg))),
       );
       if (isError(ret)) return errorFromReturn(ret);
   }

   pub fn msgrcv(msqid: u32, msgp: [*]u8, msgsz: usize, msgtyp: i64, msgflg: i32) SyscallError!usize {
       const ret = syscall5(
           uapi.syscalls.SYS_MSGRCV,
           @as(usize, msqid),
           @intFromPtr(msgp),
           msgsz,
           @as(usize, @bitCast(msgtyp)),
           @as(usize, @bitCast(@as(isize, msgflg))),
       );
       if (isError(ret)) return errorFromReturn(ret);
       return ret;
   }

   pub fn msgctl(msqid: u32, cmd: i32, buf: ?*MsqidDs) SyscallError!usize {
       const buf_addr: usize = if (buf) |b| @intFromPtr(b) else 0;
       const ret = syscall3(
           uapi.syscalls.SYS_MSGCTL,
           @as(usize, msqid),
           @as(usize, @bitCast(@as(isize, cmd))),
           buf_addr,
       );
       if (isError(ret)) return errorFromReturn(ret);
       return ret;
   }

1. Create `src/user/test_runner/tests/syscall/sysv_ipc.zig` with 12 integration tests.

   Follow existing test patterns from process_control.zig and io_mux.zig:
   - Import `syscall` module (the root.zig of user/lib/syscall)
   - Each test is `pub fn testXxx() !void`
   - Tests use `syscall.debug_print()` for diagnostics
   - Use `error.SkipTest` if a syscall returns an unexpected error suggesting the feature isn't available

   **Shared Memory Tests (4 tests):**

   a) `testShmgetCreatesSegment`:
      - Call `shmget(IPC_PRIVATE, 4096, IPC_CREAT | 0o666)` (the flags need to be cast: `@as(i32, IPC_CREAT) | 0o666`)
      - Verify returns a positive ID (> 0)
      - Clean up with `shmctl(id, IPC_RMID, null)`

   b) `testShmgetExclFails`:
      - Create segment with shmget using key=12345 and IPC_CREAT
      - Try creating again with same key and IPC_CREAT|IPC_EXCL
      - Expect EEXIST error (the error type -- check what errorFromReturn returns, likely `error.FileExists` or `error.AlreadyExists`)
      - Clean up with IPC_RMID
      - NOTE: Check the error name mapping in the test. The errno EEXIST maps to some Zig error name. Look at how other tests handle expected errors (e.g., process_control.zig or io_mux.zig).

   c) `testShmatWriteRead`:
      - Create segment: `shmget(IPC_PRIVATE, 4096, IPC_CREAT | 0o666)`
      - Attach: `shmat(id, null, 0)` -- returns pointer
      - Write data: `ptr[0] = 0xAB; ptr[1] = 0xCD;`
      - Read back and verify: `if (ptr[0] != 0xAB) return error.TestExpectedEqual;`
      - Detach: `shmdt(ptr)`
      - Destroy: `shmctl(id, IPC_RMID, null)`

   d) `testShmctlStat`:
      - Create segment of size 8192
      - Attach to get at least 1 attach
      - Call shmctl(id, IPC_STAT, &ds) to get metadata
      - Verify ds.shm_segsz >= 8192 (may be rounded up to page size)
      - Verify ds.shm_nattch == 1
      - Detach and destroy

   **Semaphore Tests (4 tests):**

   e) `testSemgetCreateSet`:
      - Call `semget(IPC_PRIVATE, 3, IPC_CREAT | 0o666)` to create set with 3 semaphores
      - Verify returns positive ID
      - Clean up with `semctl(id, 0, IPC_RMID, 0)`

   f) `testSemctlSetGetVal`:
      - Create semaphore set with 1 semaphore
      - Set value: `semctl(id, 0, SETVAL, 42)`
      - Get value: `const val = semctl(id, 0, GETVAL, 0)`
      - Verify val == 42
      - Clean up

   g) `testSemopIncrement`:
      - Create semaphore set with 1 semaphore, set initial value to 5
      - Perform semop: increment by 3 (sem_op = 3, sem_flg = 0)
      - Build SemBuf: `const sop = SemBuf{ .sem_num = 0, .sem_op = 3, .sem_flg = 0, ._pad = 0 };`
      - Call `semop(id, &[_]SemBuf{sop})`
      - Read back with GETVAL, verify == 8
      - Clean up

   h) `testSemopNowaitEagain`:
      - Create semaphore set with 1 semaphore, initial value 0
      - Try to decrement by 1 with IPC_NOWAIT: sem_op = -1, sem_flg = @truncate(@as(i32, IPC_NOWAIT))
      - NOTE: SemBuf.sem_flg is i16, IPC_NOWAIT is i32 = 0o4000 = 2048, which fits in i16. Cast appropriately.
      - Expect EAGAIN error (would block but IPC_NOWAIT set)
      - Clean up

   **Message Queue Tests (4 tests):**

   i) `testMsggetCreateQueue`:
      - Call `msgget(IPC_PRIVATE, IPC_CREAT | 0o666)`
      - Verify returns positive ID
      - Clean up with `msgctl(id, IPC_RMID, null)`

   j) `testMsgsndRecvBasic`:
      - Create queue
      - Build message buffer on stack: mtype (i64) + data bytes. The buffer layout is: 8 bytes for mtype, then data.
      - Use a local struct or raw byte array: `var buf: [8 + 16]u8 = undefined;`
      - Write mtype=1: `@as(*align(1) i64, @ptrCast(&buf)).*  = 1;`
      - Write data: `@memcpy(buf[8..][0..5], "hello");`
      - Send: `msgsnd(id, &buf, 5, 0)` -- msgsz is data size (excluding mtype header)
      - Receive: `var recv_buf: [8 + 16]u8 = undefined;`
      - `const n = msgrcv(id, &recv_buf, 16, 0, 0)` -- msgtyp=0 means any type
      - Verify n == 5
      - Verify received mtype == 1
      - Verify data matches "hello"
      - Clean up

   k) `testMsgrcvTypeFilter`:
      - Create queue
      - Send message with mtype=2, data "world"
      - Send message with mtype=1, data "hello"
      - Receive with msgtyp=1 -- should get "hello" (skip mtype=2)
      - Receive with msgtyp=0 -- should get "world" (first remaining)
      - Clean up

   l) `testMsgctlStat`:
      - Create queue
      - Send 2 messages
      - Call msgctl(id, IPC_STAT, &ds)
      - Verify ds.msg_qnum == 2
      - Clean up

   IMPORTANT TEST PATTERNS:
   - Use `catch |err| { ... }` to handle errors from syscall wrappers
   - For expected errors, match with `if (err == error.XxxError)` -- check the exact error name mapping
   - For "not implemented" detection: if shmget returns `error.NotImplemented` (ENOSYS), `return error.SkipTest;`
   - All tests should be self-contained: create IPC object, test it, destroy it
   - Do NOT use fork() for IPC tests -- test within single process (shared memory write/read works within one process)
   - Use IPC_PRIVATE for all keys to avoid collisions between tests
