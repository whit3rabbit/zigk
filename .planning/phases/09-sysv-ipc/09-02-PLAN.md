1. Create `src/kernel/ipc/sem.zig` -- semaphore set management:
   - Import: `std`, `process`, `hal`, `ipc_perm` (from same directory), `uapi`, `heap`
   - Define `Semaphore` struct: `semval: u32`, `sempid: u32` (PID of last semop)
   - Define `SemSet` struct: `id: u32`, `key: i32`, `perm: ipc_perm.IpcPerm`, `nsems: u32`, `sems: ?[*]Semaphore` (heap-allocated array), `otime: i64`, `ctime: i64`, `in_use: bool`
   - Global state: `var sem_sets: [SEMMNI]SemSet = ...` (in_use=false), `var sem_lock: SpinLock`, `var sem_seq: u16 = 0`
   - Use SEMMNI from uapi.ipc.sysv constants

   **pub fn semget(key: i32, nsems: u32, flags: i32, proc: *const process.Process) !u32:**
   - Validate nsems > 0 and nsems <= SEMMSL
   - If key == IPC_PRIVATE, allocate new set
   - Linear scan for matching key; if found check IPC_EXCL, check nsems compatibility (requested nsems must be <= existing nsems or 0), check permissions
   - Allocate: find free slot, heap-allocate Semaphore array of nsems (zero-initialized), fill metadata, return makeId

   **pub fn semop(id: u32, sops: []const uapi.ipc.sysv.SemBuf, proc: *const process.Process) !void:**
   - Find sem set by ID, verify seq match
   - Validate all operations: each sem_num < nsems, check permission (sem_op >= 0 needs write, < 0 needs read)
   - Under sem_lock, try to apply all operations atomically:
     - For each sop: if sem_op < 0, check semval >= |sem_op|; if sem_op == 0, check semval == 0
     - If any operation would block: if IPC_NOWAIT flag on that operation, return EAGAIN. Otherwise return EAGAIN (simplified MVP -- real blocking requires wait queues which are complex and rarely needed for tests).
     - If all succeed: apply operations (semval += sem_op for positive, semval -= |sem_op| for negative), update sempid for each modified semaphore
   - Update otime
   - NOTE: SEM_UNDO tracking is deferred for MVP. It requires per-process undo lists and cleanup on process exit. Add a `// TODO: SEM_UNDO tracking` comment where the flag is checked.

   **pub fn semctl(id: u32, semnum: u32, cmd: i32, arg: usize, proc: *const process.Process) !usize:**
   - Find sem set by ID
   - Switch on cmd:
     - IPC_STAT: check read permission, fill SemidDs, copy to userspace
     - IPC_SET: check isOwnerOrCreator, copy from userspace, update uid/gid/mode
     - IPC_RMID: check isOwnerOrCreator, free sems array, clear slot
     - SETVAL: validate semnum < nsems, validate arg in 0..SEMVMX, set semval, update ctime
     - GETVAL: validate semnum < nsems, return semval
     - else: return EINVAL

2. Create `src/kernel/sys/syscall/ipc/sem.zig` -- thin syscall wrappers:
   ```zig
   const uapi = @import("uapi");
   const SyscallError = uapi.errno.SyscallError;
   const user_mem = @import("user_mem");
   const sched = @import("sched");
   const process = @import("process");
   const kernel_ipc = @import("kernel_ipc");

   fn getCurrentProcess() SyscallError!*process.Process {
       const thread = sched.getCurrentThread() orelse return error.ESRCH;
       const proc_opaque = thread.process orelse return error.ESRCH;
       return @ptrCast(@alignCast(proc_opaque));
   }

   pub fn sys_semget(key: usize, nsems: usize, semflg: usize) SyscallError!usize {
       const proc = try getCurrentProcess();
       const result = kernel_ipc.sem.semget(
           @bitCast(@as(u32, @truncate(key))),
           @intCast(nsems),
           @bitCast(@as(u32, @truncate(semflg))),
           proc,
       ) catch |err| return mapIpcError(err);
       return @intCast(result);
   }

   pub fn sys_semop(semid: usize, sops_ptr: usize, nsops: usize) SyscallError!usize {
       if (nsops == 0 or nsops > uapi.ipc.sysv.SEMOPM) return error.E2BIG;
       const proc = try getCurrentProcess();

       // Copy SemBuf array from userspace
       var sops_buf: [uapi.ipc.sysv.SEMOPM]uapi.ipc.sysv.SemBuf = undefined;
       const copy_size = nsops * @sizeOf(uapi.ipc.sysv.SemBuf);
       const user_ptr = user_mem.UserPtr.from(sops_ptr);
       const dest = @as([*]u8, @ptrCast(&sops_buf))[0..copy_size];
       _ = user_ptr.copyToKernel(dest) catch return error.EFAULT;

       kernel_ipc.sem.semop(
           @intCast(semid),
           sops_buf[0..nsops],
           proc,
       ) catch |err| return mapIpcError(err);
       return 0;
   }

   pub fn sys_semctl(semid: usize, semnum: usize, cmd: usize, arg: usize) SyscallError!usize {
       const proc = try getCurrentProcess();
       return kernel_ipc.sem.semctl(
           @intCast(semid),
           @intCast(semnum),
           @bitCast(@as(u32, @truncate(cmd))),
           arg,
           proc,
       ) catch |err| return mapIpcError(err);
   }

   fn mapIpcError(err: anyerror) SyscallError {
       return switch (err) {
           error.EINVAL => error.EINVAL,
           error.EEXIST => error.EEXIST,
           error.ENOENT => error.ENOENT,
           error.EACCES => error.EACCES,
           error.ENOMEM => error.ENOMEM,
           error.ENOSPC => error.ENOSPC,
           error.EFAULT => error.EFAULT,
           error.EAGAIN => error.EAGAIN,
           error.EPERM => error.EPERM,
           error.EFBIG => error.EFBIG,
           error.ERANGE => error.ERANGE,
           error.E2BIG => error.E2BIG,
           else => error.EINVAL,
       };
   }
   ```

1. Create `src/kernel/ipc/msg.zig` -- message queue management:
   - Import: `std`, `process`, `hal`, `ipc_perm`, `uapi`, `heap`, `user_mem`
   - Define `KernelMsg` struct: `mtype: i64`, `data: []u8` (heap-allocated), `next: ?*KernelMsg` (singly-linked list)
   - Define `MsgQueue` struct: `id: u32`, `key: i32`, `perm: ipc_perm.IpcPerm`, `head: ?*KernelMsg`, `tail: ?*KernelMsg`, `qnum: usize` (message count), `qbytes: usize` (current bytes), `qbytes_max: usize` (limit, default MSGMNB), `lspid: u32`, `lrpid: u32`, `stime: i64`, `rtime: i64`, `ctime: i64`, `in_use: bool`
   - Global state: `var queues: [MSGMNI]MsgQueue = ...`, `var msg_lock: SpinLock`, `var msg_seq: u16 = 0`

   **pub fn msgget(key: i32, flags: i32, proc: *const process.Process) !u32:**
   - Same pattern as shmget/semget: IPC_PRIVATE always creates, otherwise search by key
   - No physical memory allocation needed -- queues start empty
   - Return makeId(index, seq)

   **pub fn msgsnd(id: u32, msgp: usize, msgsz: usize, msgflg: i32, proc: *const process.Process) !void:**
   - Validate msgsz <= MSGMAX
   - Find queue by ID, check write permission
   - Copy message from userspace: first read the MsgBufHeader (8 bytes for mtype), validate mtype > 0
   - Then copy data (msgsz bytes after the header)
   - Under msg_lock: check queue capacity (qbytes + msgsz <= qbytes_max). If full and IPC_NOWAIT, return EAGAIN. If full and !IPC_NOWAIT, return EAGAIN for MVP (real blocking deferred).
   - Heap-allocate KernelMsg, copy mtype and data
   - Append to tail of linked list, increment qnum and qbytes
   - Update lspid, stime

   **pub fn msgrcv(id: u32, msgp: usize, msgsz: usize, msgtyp: i64, msgflg: i32, proc: *const process.Process) !usize:**
   - Find queue by ID, check read permission
   - Under msg_lock, search for matching message:
     - msgtyp == 0: first message in queue (any type)
     - msgtyp > 0: first message with mtype == msgtyp
     - msgtyp < 0: first message with lowest mtype <= |msgtyp|
   - If no matching message: if IPC_NOWAIT return ENOMSG, else return ENOMSG for MVP (real blocking deferred)
   - Remove message from linked list, decrement qnum and qbytes
   - If message data size > msgsz: if MSG_NOERROR flag set, truncate; else return E2BIG
   - Copy to userspace: write MsgBufHeader (mtype), then data (min(actual_size, msgsz) bytes)
   - Update lrpid, rtime
   - Free KernelMsg heap allocation
   - Return actual bytes copied (excluding the mtype header)

   **pub fn msgctl(id: u32, cmd: i32, buf_ptr: usize, proc: *const process.Process) !usize:**
   - Same pattern as shmctl:
     - IPC_STAT: fill MsqidDs, copy to userspace
     - IPC_SET: update uid/gid/mode/qbytes_max
     - IPC_RMID: free all queued messages (walk linked list, free each), clear slot
   - Note: The MSG_NOERROR constant is 0o10000. Define it in sysv.zig if not already there, or define locally.
