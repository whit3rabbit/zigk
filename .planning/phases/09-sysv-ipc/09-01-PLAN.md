1. Create `src/kernel/ipc/ipc_perm.zig` -- common IPC permission checking (reused by sem.zig and msg.zig in plan 02):
   ```zig
   const process = @import("process");

   pub const IpcPerm = struct {
       key: i32,
       cuid: u32,      // Creator UID
       cgid: u32,      // Creator GID
       uid: u32,       // Owner UID
       gid: u32,       // Owner GID
       mode: u16,      // Permission bits (lower 9 bits)
       seq: u16,       // Sequence number for unique ID generation
   };

   pub const AccessMode = enum { read, write };

   pub fn checkAccess(perm: *const IpcPerm, proc: *const process.Process, mode: AccessMode) bool {
       const euid = proc.euid;
       const egid = proc.egid;
       // Root bypasses all checks
       if (euid == 0) return true;
       // Check owner permissions
       if (euid == perm.uid) {
           const need_bit: u16 = if (mode == .read) 0o400 else 0o200;
           return (perm.mode & need_bit) != 0;
       }
       // Check group permissions
       if (egid == perm.gid) {
           const need_bit: u16 = if (mode == .read) 0o040 else 0o020;
           return (perm.mode & need_bit) != 0;
       }
       // Check other permissions
       const need_bit: u16 = if (mode == .read) 0o004 else 0o002;
       return (perm.mode & need_bit) != 0;
   }

   pub fn isOwnerOrCreator(perm: *const IpcPerm, euid: u32) bool {
       return euid == 0 or euid == perm.uid or euid == perm.cuid;
   }

   /// Generate unique IPC ID from slot index and sequence number
   /// Format: (index << 16) | seq -- prevents stale ID reuse
   pub fn makeId(index: usize, seq: u16) u32 {
       return @as(u32, @intCast(index)) << 16 | @as(u32, seq);
   }

   /// Extract slot index from IPC ID
   pub fn idToIndex(id: u32) usize {
       return @intCast(id >> 16);
   }

   /// Extract sequence number from IPC ID
   pub fn idToSeq(id: u32) u16 {
       return @truncate(id);
   }
   ```

2. Create `src/kernel/ipc/shm.zig` -- shared memory segment management:
   - Import: `std`, `process` (for Process), `pmm`, `hal`, `ipc_perm` module, `uapi` (for limits)
   - Define `ShmSegment` struct with fields: `id: u32`, `key: i32`, `perm: ipc_perm.IpcPerm`, `size: usize`, `phys_pages: ?[*]u8` (physical address from PMM), `num_pages: usize`, `attach_count: u32`, `cpid: u32`, `lpid: u32`, `atime: i64`, `dtime: i64`, `ctime: i64`, `marked_for_deletion: bool`, `in_use: bool`
   - Global state: `var segments: [SHMMNI]ShmSegment = ...` (zero-initialized, in_use=false), `var shm_lock: hal.SpinLock = .{}`, `var seq_counter: u16 = 0`
   - Use `const SHMMNI = uapi.ipc.sysv.SHMMNI;` for the table size
   - `pub fn shmget(key: i32, size: usize, flags: i32, proc: *const process.Process) !u32`:
     - Acquire shm_lock
     - If key == IPC_PRIVATE, always allocate new segment
     - Otherwise linear scan for matching key. If found: check IPC_EXCL (return EEXIST), check permissions, return existing ID
     - If not found and IPC_CREAT set: allocate new segment
     - Allocate: validate size (SHMMIN <= size <= SHMMAX), find free slot, allocate zeroed pages from PMM (`pmm.allocZeroedPages`), fill metadata, bump seq_counter, return makeId(index, seq)
     - Use `std.math.add` for total pages accounting to prevent overflow
   - `pub fn shmat(id: u32, shmaddr: usize, shmflg: u32, proc: *process.Process) !usize`:
     - Acquire shm_lock, find segment by ID (idToIndex, verify seq match, verify in_use)
     - Check read or write access (SHM_RDONLY determines mode)
     - If marked_for_deletion, return EIDRM
     - Release shm_lock (VMM operations may sleep/allocate)
     - Determine virtual address: if shmaddr == 0, use proc's UserVmm.findFreeRange(). If SHM_RND, round down to page boundary.
     - Map physical pages: iterate num_pages, for each page call the paging mapPage function to map phys to virt in the process's page table. Set user=true, writable=!SHM_RDONLY, no_execute=true.
     - Create VMA via proc's UserVmm.createVma(virt_start, virt_end, prot, MAP_SHARED)
     - Insert VMA
     - Re-acquire shm_lock, increment attach_count, set lpid, atime
     - Return virtual address
     - IMPORTANT: Use `@intCast` carefully for u64/usize conversions. The UserVmm uses u64 for addresses.
   - `pub fn shmdt(shmaddr: usize, proc: *process.Process) !void`:
     - Find VMA containing shmaddr in proc's UserVmm
     - Determine which segment this VMA belongs to (scan segments for matching phys range, or store shmid in a local mapping)
     - Simpler approach for MVP: scan all segments, find one whose phys_pages match the VMA's mapped physical address at virt_start
     - Unmap pages from process page table
     - Remove VMA from proc's UserVmm
     - Acquire shm_lock, decrement attach_count, set lpid, dtime
     - If attach_count == 0 and marked_for_deletion, free physical pages and clear slot
   - `pub fn shmctl(id: u32, cmd: i32, buf_ptr: usize, proc: *const process.Process) !usize`:
     - Find segment by ID
     - Switch on cmd:
       - IPC_STAT: check read permission, fill ShmidDs from segment metadata, copy to userspace via UserPtr
       - IPC_SET: check isOwnerOrCreator, copy ShmidDs from userspace, update uid/gid/mode, update ctime
       - IPC_RMID: check isOwnerOrCreator, if attach_count==0 free pages and clear slot, else set marked_for_deletion=true and remove key (set key=-1 so new shmget won't find it)
   - Handle Spinlock type correctly: check what hal provides. The existing code uses patterns like `const held = lock.acquire(); defer held.release();` -- follow same pattern. Import the correct Spinlock from hal (check if it's `hal.SpinLock` or `@import("hal").arch.SpinLock` etc. by looking at how other kernel modules do it).

4. Create `src/kernel/sys/syscall/ipc/shm.zig` -- thin syscall wrappers:
   ```zig
   const uapi = @import("uapi");
   const SyscallError = uapi.errno.SyscallError;
   const user_mem = @import("user_mem");
   const sched = @import("sched");
   const process = @import("process");
   const kernel_ipc = @import("kernel_ipc");

   fn getCurrentProcess() SyscallError!*process.Process {
       const thread = sched.getCurrentThread() orelse return error.ESRCH;
       const proc_opaque = thread.process orelse return error.ESRCH;
       return @ptrCast(@alignCast(proc_opaque));
   }

   pub fn sys_shmget(key: usize, size: usize, shmflg: usize) SyscallError!usize {
       const proc = try getCurrentProcess();
       const result = kernel_ipc.shm.shmget(
           @bitCast(@as(u32, @truncate(key))),
           size,
           @bitCast(@as(u32, @truncate(shmflg))),
           proc,
       ) catch |err| return mapIpcError(err);
       return @intCast(result);
   }

   pub fn sys_shmat(shmid: usize, shmaddr: usize, shmflg: usize) SyscallError!usize {
       const proc = try getCurrentProcess();
       return kernel_ipc.shm.shmat(
           @intCast(shmid),
           shmaddr,
           @truncate(shmflg),
           proc,
       ) catch |err| return mapIpcError(err);
   }

   pub fn sys_shmdt(shmaddr: usize) SyscallError!usize {
       const proc = try getCurrentProcess();
       kernel_ipc.shm.shmdt(shmaddr, proc) catch |err| return mapIpcError(err);
       return 0;
   }

   pub fn sys_shmctl(shmid: usize, cmd: usize, buf: usize) SyscallError!usize {
       const proc = try getCurrentProcess();
       return kernel_ipc.shm.shmctl(
           @intCast(shmid),
           @bitCast(@as(u32, @truncate(cmd))),
           buf,
           proc,
       ) catch |err| return mapIpcError(err);
   }

   fn mapIpcError(err: anyerror) SyscallError {
       // Map kernel IPC errors to syscall errors
       return switch (err) {
           error.EINVAL => error.EINVAL,
           error.EEXIST => error.EEXIST,
           error.ENOENT => error.ENOENT,
           error.EACCES => error.EACCES,
           error.ENOMEM => error.ENOMEM,
           error.ENOSPC => error.ENOSPC,
           error.EFAULT => error.EFAULT,
           error.EIDRM => error.EIDRM,
           error.EPERM => error.EPERM,
           else => error.EINVAL,
       };
   }
   ```
