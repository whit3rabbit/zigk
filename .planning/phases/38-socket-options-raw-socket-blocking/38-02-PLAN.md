The current function requires both sockets to have SO_REUSEADDR and blocks two LISTEN sockets on the same addr:port. For SO_REUSEPORT, two LISTEN sockets ARE allowed.

Replace the function body with:
```zig
fn canReuseAddress(new_sock: *types.Socket, existing: *types.Socket) bool {
    // SO_REUSEPORT: if both sockets have it set, always allow (even two LISTENs)
    if (new_sock.so_reuseport and existing.so_reuseport) {
        return true;
    }

    // SO_REUSEADDR: both must have it
    if (!new_sock.so_reuseaddr or !existing.so_reuseaddr) {
        return false;
    }

    // TCP-specific rules (SO_REUSEADDR path)
    if (new_sock.sock_type == types.SOCK_STREAM and existing.sock_type == types.SOCK_STREAM) {
        // TIME_WAIT always allows reuse
        if (existing.tcb) |tcb| {
            if (tcb.state == tcp.TcpState.TimeWait) {
                return true;
            }
        }

        // CRITICAL: Prevent two LISTEN sockets on same addr:port with only SO_REUSEADDR
        // (SO_REUSEPORT case is handled above)
        const new_is_listen = if (new_sock.tcb) |tcb| tcb.state == tcp.TcpState.Listen else false;
        const existing_is_listen = if (existing.tcb) |tcb| tcb.state == tcp.TcpState.Listen else false;

        if (new_is_listen and existing_is_listen) {
            return false;
        }

        return true;
    }

    // UDP: allow with SO_REUSEADDR
    if (new_sock.sock_type == types.SOCK_DGRAM) {
        return true;
    }

    // Mixed types or unknown
    return false;
}
```

The key change: SO_REUSEPORT check is first and is a blanket allow, bypassing all TCP-specific restrictions including the two-listeners block.

```zig
pub fn findListeningTcbIp(local_port: u16, local_addr: IpAddr) ?*Tcb {
    var exact_match: ?*Tcb = null;
    var exact_accept: usize = std.math.maxInt(usize);
    var any_match: ?*Tcb = null;
    var any_accept: usize = std.math.maxInt(usize);

    for (listen_tcbs.items) |tcb| {
        if (tcb.local_port == local_port and tcb.state == .Listen) {
            // Look up parent socket's accept_count for FIFO dispatch
            const ac = getListenerAcceptCount(tcb);

            if (tcb.local_addr.eql(local_addr)) {
                // Exact address match -- prefer lowest accept_count
                if (ac < exact_accept) {
                    exact_match = tcb;
                    exact_accept = ac;
                }
            } else if (tcb.local_addr.isUnspecified()) {
                const family_match = switch (tcb.local_addr) {
                    .none => true,
                    .v4 => local_addr.isV4(),
                    .v6 => local_addr.isV6(),
                };
                if (family_match and ac < any_accept) {
                    any_match = tcb;
                    any_accept = ac;
                }
            }
        }
    }

    return exact_match orelse any_match;
}

/// Get accept_count from the TCB's parent socket for FIFO dispatch.
/// Returns maxInt if no parent socket (standalone TCB).
/// NOTE: Reads accept_count without the socket lock -- acceptable for
/// load-balancing heuristic (slight staleness is fine).
fn getListenerAcceptCount(tcb: *const Tcb) usize {
    const sock_idx = tcb.parent_socket orelse return std.math.maxInt(usize);
    const sock_table = socket_state_mod.getSocketTable();
    if (sock_idx >= sock_table.len) return std.math.maxInt(usize);
    const sock = sock_table[sock_idx] orelse return std.math.maxInt(usize);
    return sock.accept_count;
}
```
