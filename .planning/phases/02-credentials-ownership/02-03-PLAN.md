1. **Enhance sys_chown** (currently at line 447):
   Replace the existing simple root-only check with full POSIX permission enforcement.

   Create a helper function `chownKernel` that all chown variants can use:
   ```zig
   /// Internal: chown using a kernel-space path with POSIX permission enforcement.
   /// Implements the locked decision: file owner can chgrp to a group they belong to
   /// (supplementary or primary). Only root can change uid. Non-owner gets EPERM.
   /// Clears suid/sgid bits on ownership change (standard Linux security behavior).
   fn chownKernel(raw_path: []const u8, owner: usize, group: usize) base.SyscallError!usize {
       const alloc = heap.allocator();
       const canon_buf = alloc.alloc(u8, user_mem.MAX_PATH_LEN) catch return error.ENOMEM;
       defer alloc.free(canon_buf);

       if (raw_path.len == 0) return error.ENOENT;
       const path = canonicalizePath(raw_path, canon_buf) orelse return error.ENOENT;

       // Get current file metadata for permission checks
       const file_meta = fs.vfs.Vfs.statPath(path) orelse return error.ENOENT;
       const proc = base.getCurrentProcess();

       // Convert -1 (0xFFFFFFFF) to null for "keep current"
       const new_uid: ?u32 = if (owner == 0xFFFFFFFF or owner == 0xFFFFFFFFFFFFFFFF) null else @truncate(owner);
       const new_gid: ?u32 = if (group == 0xFFFFFFFF or group == 0xFFFFFFFFFFFFFFFF) null else @truncate(group);

       // POSIX permission enforcement (per user decision):
       // - Root (fsuid == 0) can change anything
       // - File owner can change group to a group they belong to
       // - Non-owner gets EPERM
       if (proc.fsuid != 0) {
           // Must be file owner for any chown operation
           if (proc.fsuid != file_meta.uid) {
               return error.EPERM;
           }

           // Non-root cannot change UID
           if (new_uid != null and new_uid.? != file_meta.uid) {
               return error.EPERM;
           }

           // Non-root can only change GID to a group they belong to
           if (new_gid) |gid| {
               if (!proc.isGroupMember(gid)) {
                   return error.EPERM;
               }
           }
       }

       // Perform the chown via VFS
       fs.vfs.Vfs.chown(path, new_uid, new_gid) catch |err| {
           return switch (err) {
               error.NotFound => error.ENOENT,
               error.NotSupported => error.EROFS,
               error.IOError => error.EIO,
               else => error.EIO,
           };
       };

       // Clear suid/sgid bits on ownership change (per user decision)
       // Only if uid or gid actually changed
       if (new_uid != null or new_gid != null) {
           const current_mode = file_meta.mode;
           const suid_sgid_mask: u32 = 0o6000; // S_ISUID | S_ISGID
           if (current_mode & suid_sgid_mask != 0) {
               const cleared_mode = current_mode & ~suid_sgid_mask;
               fs.vfs.Vfs.chmod(path, cleared_mode) catch {
                   // Non-fatal: ownership changed but mode clear failed
                   // This matches Linux behavior where chown succeeds even if
                   // suid/sgid clear fails on some filesystems
               };
           }
       }

       return 0;
   }
   ```

   Then replace the existing `sys_chown` body to delegate:

3. **Implement sys_fchown** (new):
   Operates on fd directly via FileOps.chown:
   ```zig
   /// sys_fchown (93/55) - Change file owner and group by file descriptor
   /// Per user decision: uses direct FileOps.chown to avoid TOCTOU race (no path extraction)
   pub fn sys_fchown(fd_num: usize, owner: usize, group: usize) base.SyscallError!usize {
       const table = base.getGlobalFdTable();
       const fd_u32 = std.math.cast(u32, fd_num) orelse return error.EBADF;
       const file_desc = table.get(fd_u32) orelse return error.EBADF;

       const proc = base.getCurrentProcess();
       const new_uid: ?u32 = if (owner == 0xFFFFFFFF or owner == 0xFFFFFFFFFFFFFFFF) null else @truncate(owner);
       const new_gid: ?u32 = if (group == 0xFFFFFFFF or group == 0xFFFFFFFFFFFFFFFF) null else @truncate(group);

       // Try direct FileOps.chown first (preferred, avoids TOCTOU)
       if (file_desc.ops.chown) |chown_fn| {
           // Permission check via file metadata from stat
           if (file_desc.ops.stat) |stat_fn| {
               var stat_buf: @import("uapi").stat.Stat = undefined;
               const stat_ret = stat_fn(file_desc, @ptrCast(&stat_buf));
               if (stat_ret >= 0) {
                   // POSIX permission enforcement
                   if (proc.fsuid != 0) {
                       if (proc.fsuid != stat_buf.st_uid) return error.EPERM;
                       if (new_uid != null and new_uid.? != stat_buf.st_uid) return error.EPERM;
                       if (new_gid) |gid| {
                           if (!proc.isGroupMember(gid)) return error.EPERM;
                       }
                   }
               }
           }

           const result = chown_fn(file_desc, new_uid, new_gid);
           if (result < 0) {
               // Convert negative errno to error
               return error.EIO;
           }
           return 0;
       }

       // Fallback: if file has a path, use path-based chown
       // For files opened with a known path, extract and use chownKernel
       // Most kernel file descriptors don't store their path, so this
       // is a best-effort fallback. Return ENOSYS if neither works.
       // In practice, SFS files opened via open() go through VFS which
       // gives them path info stored in the file descriptor context.

       // For now, if no FileOps.chown exists, return ENOSYS
       // (fchown requires the fd's filesystem to support it)
       return error.ENOSYS;
   }
   ```

4. **Implement sys_fchownat** (new):
   Follows the existing *at syscall pattern from sys_fchmodat/sys_mkdirat:
   ```zig
   /// sys_fchownat (260/54) - Change file owner/group relative to directory fd
   /// Supports AT_FDCWD, AT_SYMLINK_NOFOLLOW, AT_EMPTY_PATH per user decision
   pub fn sys_fchownat(dirfd: usize, path_ptr: usize, owner: usize, group: usize, flags: usize) base.SyscallError!usize {
       const AT_SYMLINK_NOFOLLOW: usize = 0x100;
       const AT_EMPTY_PATH: usize = 0x1000;

       // AT_EMPTY_PATH: operate on dirfd itself (reuse fchown path per Claude's discretion - DRY)
       if (flags & AT_EMPTY_PATH != 0) {
           return sys_fchown(dirfd, owner, group);
       }

       const alloc = heap.allocator();
       const path_buf = alloc.alloc(u8, user_mem.MAX_PATH_LEN) catch return error.ENOMEM;
       defer alloc.free(path_buf);

       const raw_path = user_mem.copyStringFromUser(path_buf, path_ptr) catch |err| {
           if (err == error.NameTooLong) return error.ENAMETOOLONG;
           return error.EFAULT;
       };

       if (raw_path.len == 0) return error.ENOENT;

       // Handle absolute paths directly (bypass dirfd)
       if (raw_path[0] == '/') {
           return chownKernel(raw_path, owner, group);
       }

       // Resolve path relative to dirfd
       const resolved_buf = alloc.alloc(u8, user_mem.MAX_PATH_LEN) catch return error.ENOMEM;
       defer alloc.free(resolved_buf);
       const resolved = fd_syscall.resolvePathAt(dirfd, raw_path, resolved_buf) catch |err| return err;

       // AT_SYMLINK_NOFOLLOW: use nofollow variant
       // (Currently identical since VFS doesn't follow symlinks, but correct API)
       _ = flags & AT_SYMLINK_NOFOLLOW; // Acknowledged but same codepath
       return chownKernel(resolved, owner, group);
   }
   ```
